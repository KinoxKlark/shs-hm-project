event : sport_positif
{
	start
	perso = ?a
	descr = ?a aime le sport
	cond = like(sport,?a)
}

post : sport_positif1
{
	start
	event = sport_positif(?a)
	type = publication(?a)
	text = "J'aime garder la forme en faisant du sport."
	modif = ?&.sport+
}
