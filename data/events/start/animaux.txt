event : animaux_positif
{
	start
	perso = ?a
	descr = ?a aime les animaux
	cond = like(animaux,?a)
}

post : animaux_positif1
{
	start
	event = animaux_positif(?a)
	type = publication(?a)
	text = "Je suis allé en forêt observer les animaux, ils sont tout aussi fascinant les uns que les autres !"
	modif = ?&.animaux+
}
