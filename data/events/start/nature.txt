event : nature_positif
{
	start
	perso = ?a
	descr = ?a aime la nature
	cond = like(nature,?a)
}

post : nature_positif1
{
	start
	event = nature_positif(?a)
	type = publication(?a)
	text = "J'aime la nature !"

}
