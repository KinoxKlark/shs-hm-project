event : cuisine_positif
{
	start
	perso = ?a
	descr =  ?a aime la cuisine
	cond = like(cuisine,?a)
}

post : cuisine_positif1
{
	start
	event = cuisine_positif(?a)
	type = photo(?a,hot)
	text = "Petit gâteau que j'ai fait en suivant la recette de ma grand-mère."
	modifs = ?&.cuisine+
}

post : cuisine_positif2
{
	start
	event = cuisine_positif(?a)
	type = publication(?a)
	text = "J'aime bien recevoir des amis pour manger chez moi, car celà me permet de tester des nouvelles recettes !"
	modifs = ?&.cuisine+
}

post : cuisine_positif3

{
	start
	perso = ?a
	descr =  ?a n'aime pas la cuisine
	cond = dislike(cuisine,?a)
}
