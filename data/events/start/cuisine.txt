event : cuisine_positif
{
	start
	perso = ?a
	descr =  ?a aime la cuisine
	cond = like(cuisine,?a)
}

post : cuisine_positif1
{
	start
	event = cuisine_positif(?a)
	type = publication(?a)
	text = "J'aime vraiment cuisiner ! Je cuisine dès que j'en ai le temps !"

}
