event : humour_positif
{
	start
	perso = ?a
	descr = ?a a de l'humour
	cond = like(humour,?a)
}

post : humour_positif1
{
	start
	event = humour_positif(?a)
	type = publication(?a)
	text = "Poisson d'avril !"
	modif = ?&.humour+
}
