event : complot_positif
{
	start
	perso = ?a
	descr = ?a est pour les theorie du complot
	cond = like(complot,?a)
}

post : complot_positif1
{
	start
	event = complot_positif(?a)
	type = publication(?a)
	text = "Je suis sûr qu'un complot se prépare !"
	modif = ?&.complot+
}

event : complot_negatif
{
	start
	perso = ?a
	descr = ?a ne croit pas aux théories du complot
	cond = dislike(complot,?a)
}
