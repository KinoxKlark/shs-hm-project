event : respectueux_positif
{
	start
	perso = ?a
	descr = ?a est respectueux
	cond = like(respectueux,?a)
}

post : respectueux_positif1
{
	start
	event = respectueux_positif(?a)
	type = publication(?a)
	text = "C'est malheureusement les personnes les plus honnêtes qui se font avoir par les autres ..."
	modif = ?&.respectueux+
}
