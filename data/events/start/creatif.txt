event : creatif_positif
{
	start
	perso = ?a
	descr = ?a est creatif
	cond = like(creatif,?a)
}

post : creatif_positif1
{
	start
	event = creatif_positif(?a)
	type = photo(?a,turf)
	text = "Nouvelle sculpture."
	modif = ?&.creatif+
}

post : creatif_positif2
{
	start
	event = creatif_positif(?a)
	type = publication(?a)
	text = "Ma prof de dessin était contente de moi aujourd'hui"
	modif = ?&.creatif+
}

event : creatif_negatif
{
	start
	perso = ?a
	descr = ?a n'est pas creatif
	cond = dislike(creatif,?a)
}
