event : bordu1
{
	perso = ?a , ?b
	descr = ?a propose à ?b d'aller au bordu
	cond = [friend(?a,?b)]
	modif = ?a.nature+
}

event : bordu2
{
	perso = ?a , ?b
	descr = ?a et ?b vont au bordu ensemble.
	cond = event(bordu1, ?a, ?b)
	modif = [?a.lien.?b+, ?b.nature+]
}

post : bordu01
{
  event = bordu2(?a,?b)
  type = publication(?a)
  text = "?a publie une photo au bord d'un lac avec ?b"
}

post : bordu02
{
  event = bordu2(?a,?b)
  type = publication(?b)
  text = "?b publie une photo au bord d'un lac avec ?a"
}

# J'ai mis que l'obs a peu dans nature pour etre sur qu il est pas allé au bordu aussi pour l'instant ...j vais modif
event : reaction_jalou_bordu01
{
	perso = ?obs, ?a , ?b
	descr = ?obs est jaloux de ne pas avoir été invité au bordu par ?a
	cond = [saw(bordu01, ?obs, ?a,?b), low(rationnel,?obs) , friend(?obs,?a) , enemy(?obs,?b)]
	modif = [?obs.lien.?b-- , ?obs.rationnel-]
}

}

{
	perso = ?obs, ?a , ?b
	descr = ?obs trouve ça cool la ballade au bordu de ?b et ?a

	modif = [?obs.lien.?a+ ,?obs.lien.?b+, ?obs.nature+]
}
