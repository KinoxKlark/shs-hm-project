event : choppe_grippe_aviaire
{
	perso = ?a
	descr = ?a  attrape la grippe aviaire
	cond =  high(animaux,?a)
}

post : malade_grippe
{
	event = choppe_grippe_aviaire(?a)
	type  = publication(?a)
	text = "J'ai attrapé la grippe aviaire."
	modifs = ?&.animaux-
}
